# Toma el data frame `matriz` (cargado desde un archivo Excel previamente)
# y lo transforma en una matriz de adyacencia numrica.

# Paso 1: column_to_rownames(var = colnames(.)[1])
#   - Convierte la primera columna (por ejemplo, "Concepto") en los nombres de las filas.
#   - Esto es til si esa columna representa los nodos (conceptos) de la red.

# Paso 2: mutate(across(everything(), as.numeric))
#   - Aplica `as.numeric()` a todas las columnas del data frame.
#   - Esto asegura que todos los valores sean interpretados como nmeros (por ejemplo, 0s y 1s),
#     y no como texto o factores. Es esencial para que la matriz sea vlida como adyacencia.

# Paso 3: as.matrix()
#   - Convierte el data frame limpio en una matriz (tipo base de R).
#   - Esta matriz se puede usar directamente para crear grafos o realizar operaciones algebraicas.

# Resultado:
# `matriz_adj` es una matriz cuadrada con nombres de fila y columna, 
# representando las conexiones (adyacencias) entre conceptos en formato binario o numrico.

matriz_adj<-matriz%>%
  column_to_rownames(var = colnames(.)[1])%>%                         
  mutate(across(everything(), as.numeric)) %>%  
  as.matrix()

# Reemplaza todos los valores NA (valores perdidos) en la matriz de adyacencia `matriz_adj` por ceros.

# lista para ser usada en anlisis de redes o visualizaciones (por ejemplo, con igraph o ggraph).

matriz_adj[is.na(matriz_adj)] <- 0

# Crea un grafo dirigido (objeto de clase `igraph`) a partir de la matriz de adyacencia `matriz_adj`.

# Argumentos:
# - matriz_adj: matriz cuadrada con valores numricos (por ejemplo, 0 y 1),
#   donde cada celda [i, j] indica si existe una conexin desde el nodo i al nodo j "j necesita de i para ser comprendido" .
#
# - mode = "directed":
#   especifica que las conexiones representadas en la matriz son dirigidas,
#   es decir, que una arista desde A hacia B no implica necesariamente una desde B hacia A.

# Resultado:
# - `g` es un objeto de clase `igraph` que representa el grafo dirigido.
# - Puede ser usado para anlisis de redes, visualizacin, extraccin de componentes,
#   deteccin de caminos, ciclos, grados, etc.

# Nota:
# - Los nombres de los nodos sern tomados desde los nombres de fila/columna de `matriz_adj`, si estn presentes.

g <- graph_from_adjacency_matrix(matriz_adj, mode = "directed")

# Calcula el cierre transitivo de la matriz de adyacencia:
# - Suma todas las potencias de la matriz hasta n (longitud mxima de caminos)
# - Luego binariza: 1 si existe al menos un camino, 0 si no


# Elevarla sucesivamente para obtener rutas indirectas
n <- nrow(matriz_adj)
closure <- matriz_adj
for (k in 2:n) {
  closure <- closure + (matriz_adj %^% k)
}

# Binarizar (1 si hay camino, 0 si no)
closure_binaria <- (closure > 0) * 1

# Dependencias directas: Guarda la versin original de la matriz, con conexiones explcitas (1 si hay arco directo, 0 si no).
directas <- matriz_adj 

# Cierre transitivo binario previamente calculado: Representa todas las conexiones posibles entre nodos, 
#ya sean directas o indirectas (de cualquier longitud).
transitivo <- closure_binaria

# Diferencia entre cierre y original: transitivas puras: Marca como TRUE aquellas dependencias que solo 
#existen por transitividad (es decir, no son aristas directas, pero s alcanzables va otros nodos).
transitivas_puras <- (transitivo - directas) > 0

# Solo dependencias directas (sin transitivas):  Mantiene solo las conexiones que son directas y no redundantes, 
#descartando las derivadas de rutas ms largas.
solo_directas <- directas * (!transitivas_puras)

# Resultado final: Resultado: una matriz optimizada que refleja solo las dependencias esenciales, 
#lista para modelar precedencias sin ruido.
matriz_sin_redundancia <- solo_directas
